    Kotlin学习¶

    一、函数¶

    fun sum(x:Int,y:Int):Int {  // 第一种方式
    return x + y
}
fun sum2(x:Int,y:Int) = x+y // 第二种方式
val sumLambda: (Int,Int) -> Int = {x,y -> x+y} // 花括号为匿名函数

    fun vars(vararg l:Int){
    for (i in l){
        print(i)
    }
}

    vararg类似于python的解包操作

    def vars(*lists):
    for i in lists:
        print(i,end="")

    二、变量¶

    var用于定义变量，形式为：

    var <标识符> : <类型> = <初始值>

    val用于定义不可变变量，只能赋值一次

    val <标识符> : <类型> = <初始值>

    变量的操作类似于golang语言

    val a: Int = 1
val b = 1       // 系统自动推断变量类型为Int
val c: Int      // 如果不在声明时初始化则必须提供变量类型
c = 1           // 明确赋值


var x = 5        // 系统自动推断变量类型为Int
x += 1           // 变量可修改

    三、循环¶

    for (i in 1..10 step 2){ // [1,10]
    print(i)
}
for (i in 4 downTo 1 step 2){ //downTo 
    print(i)
}
for (i in 1 until 10){ // [1,10)
    print(i)
}
for (item: Int in ints) { // 和python一样可以遍历任意可迭代对象
    // ……
}
for ((index,value) in array.withIndex()){ // array.withIndex()类似于python中的enumerate(array)
    println("$index 索引上的值为$value")
}

    class 类名 [可见性修饰符] [注解] [constructor] (Params){

  }

class Person(name: String) { …… } // 没有可见性修饰符和注释可以简写成这样
class Test(name: String){
    val name_class: String = name //这个才是类属性
    init{
        // 初始化代码块
    }
}
class Test(var name: String){ // 在主构造函数中声明类属性

}
class Test{                   // 次构造函数 注意Test后面不要加括号
    var name: String = ""
    constructor(name: String){
        this.name = name
    }
}
class Test(name: String){  // 次构造函数
    var name: String = name
    var age: Int = 1
    constructor(name: String,age: Int) : this(name){  // 注意这个this
        this.age = age
    }
}
